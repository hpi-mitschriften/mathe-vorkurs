\begin{itemize}
    \item Spezialfall der Proportionalität
    \item Zwei größen:\
    \begin{itemize}
        \item Prozente
        \item andere Größe
    \end{itemize}
    \item $1\% = \frac{1}{100}$
    \item $1\permil = \frac{1}{1000}$
    \item Grundwert $G \hat{=} 100\%$, Prozentwert $W \hat{=} p\%$
    $$\frac{G}{100\%} = \frac{W}{p\%}$$
\end{itemize}
